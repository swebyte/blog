# Install postgrest
choco install postgrest


# Pull the latest PostgreSQL Docker image
docker pull postgres:latest

# Start a new PostgreSQL container with the setup.sql script
docker run --name my_postgres \
  -e POSTGRES_USER=admin \
  -e POSTGRES_PASSWORD=admin \
  -e POSTGRES_DB=mydb \
  -v c:/git/pgall/database:/docker-entrypoint-initdb.d \
  -p 5432:5432 \
  -d postgres:latest

# Connect to the PostgreSQL database inside the container
docker exec -it my_postgres psql -U admin -d mydb

# Check tables and roles in the PostgreSQL shell
\dt
\du

# Connection string to access the database from the host
postgres://admin:admin@localhost:5432/mydb
